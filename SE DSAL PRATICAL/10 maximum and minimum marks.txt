#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
// Function to heapify for Max Heap
void maxHeapify(vector<int>& arr, int n, int i) {
int largest = i;
int l = 2*i + 1;
int r = 2*i + 2;
if (l < n && arr[l] > arr[largest])
largest = l;
if (r < n && arr[r] > arr[largest])
largest = r;
if (largest != i) {
swap(arr[i], arr[largest]);
maxHeapify(arr, n, largest);
}
}
// Function to heapify for Min Heap
void minHeapify(vector<int>& arr, int n, int i) {
int smallest = i;
int l = 2*i + 1;
int r = 2*i + 2;
if (l < n && arr[l] < arr[smallest])
smallest = l;
if (r < n && arr[r] < arr[smallest])
smallest = r;
if (smallest != i) {
swap(arr[i], arr[smallest]);
minHeapify(arr, n, smallest);
}
}
int main() {
int n;
cout << "Enter number of students: ";
cin >> n;
vector<int> marks(n);
cout << "Enter marks obtained by students:\n";
for (int i = 0; i < n; i++) {
cin >> marks[i];}
// Copy marks for separate heap builds
vector<int> maxHeap = marks;
vector<int> minHeap = marks;
// Build Max Heap
for (int i = n / 2 - 1; i >= 0; i--)
maxHeapify(maxHeap, n, i);
// Build Min Heap
for (int i = n / 2 - 1; i >= 0; i--)
minHeapify(minHeap, n, i);
cout << "\nMaximum marks: " << maxHeap[0];
cout << "\nMinimum marks: " << minHeap[0];
return 0;
}
